<script>
  /* ***** 👇 فقط این قسمت را ویرایش کنید 👇 ***** */
  const MENU_ITEMS = [
    { id:1, name:"لاته", price:125000, category:"قهوه‌های گرم", img:"https://picsum.photos/seed/latte/200/200", desc:"اسپرسو + شیر فوم‌دار" },
    { id:2, name:"آمریکانو", price:95000, category:"قهوه‌های گرم", img:"https://picsum.photos/seed/americano/200/200", desc:"اسپرسو + آب داغ" },
    { id:3, name:"کاپوچینو", price:115000, category:"قهوه‌های گرم", img:"https://picsum.photos/seed/capu/200/200", desc:"اسپرسو + فوم بیشتر" },
    { id:4, name:"آیس لاته", price:135000, category:"قهوه‌های سرد", img:"https://picsum.photos/seed/icelatte/200/200", desc:"سرو شده با یخ" },
    { id:5, name:"موکا", price:145000, category:"قهوه‌های گرم", img:"https://picsum.photos/seed/mocha/200/200", desc:"اسپرسو + سس شکلات" },
    { id:6, name:"چیزکیک", price:175000, category:"دسر", img:"https://picsum.photos/seed/cheese/200/200", desc:"چیزکیک خانگی" },
  ];
  /* ***** ☝️ این بخش را با آیتم‌های واقعی خودتان جایگزین کنید ☝️ ***** */

  const gridEl = document.getElementById('grid');
  const searchEl = document.getElementById('search');
  const catEl = document.getElementById('category');
  const updatedEl = document.getElementById('updated');
  const printBtn = document.getElementById('printBtn');
  const toTopBtn = document.getElementById('toTop');

  // دسته‌بندی‌ها
  const categories = Array.from(new Set(MENU_ITEMS.map(i=>i.category))).sort();
  categories.forEach(c=>{
    const o=document.createElement('option'); o.value=c; o.textContent=c; catEl.appendChild(o);
  });

  function render(items){
    gridEl.innerHTML = '';
    if(!items.length){
      const p = document.createElement('p'); p.className='muted'; p.textContent='آیتمی یافت نشد.'; gridEl.appendChild(p); return;
    }
    items.forEach(i=>{
      const card = document.createElement('div'); card.className='card';
      const img  = document.createElement('img'); img.className='thumb'; img.loading='lazy'; img.alt=i.name; img.src=i.img;
      const block = document.createElement('div');
      const title = document.createElement('div'); title.className='title'; title.textContent = i.name;
      const desc = document.createElement('div'); desc.className='desc'; desc.textContent = i.desc || '';
      const row = document.createElement('div'); row.className='row';
      const price = document.createElement('div'); price.className='price'; price.textContent = formatPrice(i.price);
      const cat = document.createElement('div'); cat.className='cat'; cat.textContent = i.category;
      row.append(price, cat);
      block.append(title, desc, row);
      card.append(img, block);
      gridEl.appendChild(card);
    });
  }

  function applyFilter(){
    const q = (searchEl.value||'').trim();
    const c = catEl.value;
    const filtered = MENU_ITEMS.filter(i=>{
      const inCat = c==='all' || i.category===c;
      const inTxt = !q || [i.name,i.desc,i.category].filter(Boolean).some(t=>t.includes(q));
      return inCat && inTxt;
    });
    render(filtered);
  }

  function formatPrice(num){
    if(typeof num!=='number') return '';
    return num.toLocaleString('fa-IR') + ' تومان';
  }

  searchEl.addEventListener('input', applyFilter);
  catEl.addEventListener('change', applyFilter);
  printBtn.addEventListener('click', ()=>window.print());
  toTopBtn.addEventListener('click', ()=>window.scrollTo({top:0,behavior:'smooth'}));

  const d = new Date();
  updatedEl.textContent='آخرین به‌روزرسانی: '+d.toLocaleDateString('fa-IR')+'، '+d.toLocaleTimeString('fa-IR',{hour:'2-digit',minute:'2-digit'});

  render(MENU_ITEMS);
</script>
